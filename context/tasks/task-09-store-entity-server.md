# Task 09 — Store Core, EntityServer, and EntityClient

## Scope

Implement the full Store Layer:

1. `Store<S>` — the central data structure with `HashMap<AnyEntityRef, StoreEntity>`, change-tracking sets, and all store operations
2. `EntityServer` — tokio actor that exclusively owns `Store<S>`, processes messages
3. `EntityClient` — stateless async wrapper over the message protocol
4. `StoreMessage`, `StoreRequest`, `StoreCommand`, `StoreResponse` — internal message protocol
5. `StoreEntity::all_refs()` — cross-entity ref collection (generated by derive macro update)
6. `get_or_load` — real accessor implementation (replaces stub from Task 03)
7. `ensure_mutable` — real setter pre-condition (replaces stub from Task 03)
8. Cross-entity primitives from Task 07 (`ref_exists`, `has_ref`, cycle detection) — replace stubs

This task uses `tokio` as the async runtime. `StoreEntity` (the store-level enum from Task 04) is used as the value type in `entities`.

### `remove`

```rust
fn remove(&mut self, any_ref: &AnyEntityRef) -> Result<StoreEntity, StoreOpError> {
    if self.checked_out.contains(any_ref) {
        return Err(StoreOpError::CheckedOut);
    }
    if !self.entities.contains_key(any_ref) {
        // Already removed or never existed — no-op
        if self.removed.contains(any_ref) { return Err(StoreOpError::AlreadyRemoved); }
        return Err(StoreOpError::NotFound);
    }
    let entity = self.entities.remove(any_ref).unwrap();

    // Transition logic
    if self.added.remove(any_ref) {
        // add-then-remove: net no-op; don't add to removed
    } else {
        self.modified.remove(any_ref);
        self.removed.insert(any_ref.clone());
    }
    Ok(entity)
}
```

### `resolve` (async — may hit substrate)

async fn resolve(&mut self, any_ref: AnyEntityRef) -> Result<StoreEntity, ResolveError> {
    if let Some(entity) = self.entities.get(&any_ref) {
        return Ok(entity.clone());
    }
    // Cache miss — check substrate
    match self.substrate.exists(&any_ref).await {
        Ok(true) => {}
        Ok(false) => return Err(ResolveError::NotFound),
        Err(e) => return Err(ResolveError::SubstrateError(e)),

    self.entities.insert(any_ref.clone(), stub.clone());

}

    if self.checked_out.contains(&any_ref) {

    }
    let entity = self.entities.get(&any_ref)

    self.checked_out.insert(any_ref);

### `commit`

```rust
async fn commit(&mut self, entity: StoreEntity, any_ref: AnyEntityRef) -> Result<(), CommitError> {
    if !self.checked_out.contains(&any_ref) {
        return Err(CommitError::NotCheckedOut);
    }

    // Validation (cross-entity only for modified; full for new)
    let is_new = self.added.contains(&any_ref);
    // ... run_validations call (see Task 07 integration)

    // Merge dirty fields
    if let Some(store_entity) = self.entities.get_mut(&any_ref) {
        entity.merge_dirty_into(store_entity); // delegates to TrackedX::merge_dirty_into
    }

    // Update change lists
    let has_dirty = entity.has_dirty_fields();
    if has_dirty && !is_new {
        self.modified.insert(any_ref.clone());
    }
    if is_new && has_dirty {
        // Clear dirty flags on added entities
        if let Some(e) = self.entities.get_mut(&any_ref) { e.reset_dirty(); }
    }

    self.checked_out.remove(&any_ref);
    Ok(())
}
```

### `undo_commit`

```rust
fn undo_commit(&mut self, any_ref: &AnyEntityRef) -> Result<(), UndoError> {
    if self.checked_out.contains(any_ref) { return Err(UndoError::WrongState); }
    if self.added.remove(any_ref) {
        self.entities.remove(any_ref);
        return Ok(());
    }
    if self.modified.remove(any_ref) {
        // Replace with stub
        let stub = StoreEntity::make_stub(any_ref);
        self.entities.insert(any_ref.clone(), stub);
        return Ok(());
    }
    Err(UndoError::WrongState)
}
```

fn unload(&mut self, any_ref: &AnyEntityRef) -> Result<(), UndoError> {
    if self.checked_out.contains(any_ref)
        || self.added.contains(any_ref)
        || self.modified.contains(any_ref)
        || self.removed.contains(any_ref)

        self.entities.insert(any_ref.clone(), stub);
        Ok(())

        Err(UndoError::WrongState)

async fn load(&mut self, any_ref: AnyEntityRef, fields: Vec<String>) -> Result<StoreEntity, LoadError> {
    // Step 0: short-circuit if all requested fields already initialized
    // Step 1: fetch from substrate
    let partial = self.substrate.load(&any_ref, &fields.iter().map(|s| s.as_str()).collect::<Vec<_>>()).await
        .map_err(|e| LoadError::Substrate(e))?;

    // Step 2: all_refs() expansion
    let refs = partial.all_refs();
    for r in refs {
        if !self.entities.contains_key(&r) {
            if let Ok(true) = self.substrate.exists(&r).await {
                let stub = StoreEntity::make_stub(&r);
                self.entities.insert(r, stub);
            }
        }
    }

    // Step 3: validate
    // ... run_validations on partial with all kinds

    // Step 4: merge into cached entity
    if let Some(cached) = self.entities.get_mut(&any_ref) {
        partial.initialize_into(cached); // write-once merge
    }

    Ok(self.entities.get(&any_ref).unwrap().clone())
}

### `persist` (async)

```rust
async fn persist(&mut self) -> Result<(), PersistError> {
    if !self.checked_out.is_empty() {
        return Err(PersistError::PendingCheckouts);
    }

    let changes: Vec<EntityChange> = self.changes().collect();
    self.substrate.atomic_persist(&changes).await
        .map_err(|errors| PersistError::SubstrateError(errors))?;

    // Reset on success
    for any_ref in &self.modified {
        if let Some(e) = self.entities.get_mut(any_ref) { e.reset_dirty(); }
    }
    self.added.clear();
    self.modified.clear();
    self.removed.clear();
    Ok(())
}

fn changes(&self) -> impl Iterator<Item = EntityChange<'_>> {
    let added   = self.added.iter().map(|r| EntityChange::Added(self.entities.get(r).unwrap()));
    let modified = self.modified.iter().map(|r| {
        let e = self.entities.get(r).unwrap();
        EntityChange::Modified(e, e.dirty_fields())
    });
    let removed = self.removed.iter().map(|r| EntityChange::Removed(r.clone()));
    added.chain(modified).chain(removed)
}
```

## EntityServer

```rust
use std::sync::OnceLock;
use std::cell::RefCell;
use tokio::sync::mpsc;

static GLOBAL_SENDER: OnceLock<mpsc::Sender<StoreMessage>> = OnceLock::new();

thread_local! {
    static OVERRIDE_SENDER: RefCell<Option<mpsc::Sender<StoreMessage>>> = RefCell::new(None);
}

pub struct EntityServer;

impl EntityServer {
    pub fn init(substrate: impl Substrate) {
        let (tx, rx) = mpsc::channel(32);
        let store = Store::new(substrate);
        tokio::spawn(async move { store.run(rx).await });
        GLOBAL_SENDER.set(tx).expect("EntityServer already initialized");
    }

    pub fn sender() -> mpsc::Sender<StoreMessage> {
        OVERRIDE_SENDER.with(|o| o.borrow().clone())
            .unwrap_or_else(|| GLOBAL_SENDER.get()
                .expect("EntityServer not initialized").clone())
    }

    /// Test helper: scoped thread-local override.
    pub async fn with_test<F, Fut>(substrate: impl Substrate, f: F)
    where F: FnOnce() -> Fut, Fut: std::future::Future<Output = ()>
    {
        let (tx, rx) = mpsc::channel(32);
        tokio::spawn(Store::new(substrate).run(rx));
        OVERRIDE_SENDER.with(|o| *o.borrow_mut() = Some(tx));
        f().await;
        OVERRIDE_SENDER.with(|o| *o.borrow_mut() = None);
    }
}

impl<S: Substrate> Store<S> {
    async fn run(mut self, mut rx: mpsc::Receiver<StoreMessage>) {
        while let Some(msg) = rx.recv().await {
            match msg {
                StoreMessage::Request { request, reply } => {
                    let _ = reply.send(self.handle(request).await);
                }
                StoreMessage::Command(cmd) => {
                    self.execute(cmd);
                }
            }
        }
    }

    async fn handle(&mut self, req: StoreRequest) -> Result<StoreResponse, StoreError> {
        match req {
            StoreRequest::Resolve { any_ref } =>
                self.resolve(any_ref).await
                    .map(StoreResponse::Entity)
                    .map_err(StoreError::from),
            StoreRequest::Checkout { any_ref } =>
                self.checkout(any_ref)
                    .map(StoreResponse::Entity)
                    .map_err(StoreError::from),
            StoreRequest::Commit { entity, any_ref } =>
                self.commit(entity, any_ref).await
                    .map(|_| StoreResponse::Unit)
                    .map_err(StoreError::from),
            StoreRequest::Remove { any_ref } =>
                self.remove(&any_ref)
                    .map(StoreResponse::Entity)
                    .map_err(StoreError::from),
            StoreRequest::Persist =>
                self.persist().await
                    .map(|_| StoreResponse::Unit)
                    .map_err(StoreError::from),
            StoreRequest::Load { any_ref, fields } =>
                self.load(any_ref, fields).await
                    .map(StoreResponse::Entity)
                    .map_err(StoreError::from),
            StoreRequest::UndoCommit { any_ref } =>
                self.undo_commit(&any_ref)
                    .map(|_| StoreResponse::Unit)
                    .map_err(StoreError::from),
            StoreRequest::Unload { any_ref } =>
                self.unload(&any_ref)
                    .map(|_| StoreResponse::Unit)
                    .map_err(StoreError::from),
        }
    }

    fn execute(&mut self, cmd: StoreCommand) {
        match cmd {
            StoreCommand::Insert(entity) => {
                let any_ref = entity.any_ref(); // generated method
                if self.removed.remove(&any_ref) {
                    self.modified.insert(any_ref.clone());
                } else {
                    self.added.insert(any_ref.clone());
                }
                self.entities.insert(any_ref, entity);
            }
            StoreCommand::UndoCheckout { any_ref } => {
                self.checked_out.remove(&any_ref);
            }
        }
    }
}
```

## EntityClient

```rust
pub struct EntityClient;

impl EntityClient {
    pub async fn resolve(any_ref: AnyEntityRef) -> Result<StoreEntity, StoreError> {
        match request(StoreRequest::Resolve { any_ref }).await? {
            StoreResponse::Entity(e) => Ok(e),
            _ => unreachable!(),
        }
    }

    pub async fn insert(entity: StoreEntity) -> Result<(), StoreError> {
        send(StoreCommand::Insert(entity)).await
    }

    pub async fn remove(any_ref: AnyEntityRef) -> Result<StoreEntity, StoreError> {
        match request(StoreRequest::Remove { any_ref }).await? {
            StoreResponse::Entity(e) => Ok(e),
            _ => unreachable!(),
        }
    }

    pub async fn checkout(any_ref: AnyEntityRef) -> Result<StoreEntity, CheckoutError> {
        match request(StoreRequest::Checkout { any_ref }).await
            .map_err(|_| CheckoutError::StoreUnavailable)?
        {
            StoreResponse::Entity(e) => Ok(e),
            _ => unreachable!(),
        }
    }

    pub async fn persist() -> Result<(), PersistError> {
        match request(StoreRequest::Persist).await
            .map_err(|_| PersistError::StoreUnavailable)?
        {
            StoreResponse::Unit => Ok(()),
            _ => unreachable!(),
        }
    }

    pub async fn undo_commit(any_ref: AnyEntityRef) -> Result<(), UndoError> {
        match request(StoreRequest::UndoCommit { any_ref }).await
            .map_err(|_| UndoError::StoreUnavailable)?
        {
            StoreResponse::Unit => Ok(()),
            _ => unreachable!(),
        }
    }

    pub async fn unload(any_ref: AnyEntityRef) -> Result<(), UndoError> {
        match request(StoreRequest::Unload { any_ref }).await
            .map_err(|_| UndoError::StoreUnavailable)?
        {
            StoreResponse::Unit => Ok(()),
            _ => unreachable!(),
        }
    }
}

async fn request(req: StoreRequest) -> Result<StoreResponse, StoreError> {
    let (tx, rx) = tokio::sync::oneshot::channel();
    EntityServer::sender()
        .send(StoreMessage::Request { request: req, reply: tx })
        .await
        .map_err(|_| StoreError::Unavailable)?;
    rx.await.map_err(|_| StoreError::Unavailable)?
}

async fn send(cmd: StoreCommand) -> Result<(), StoreError> {
    EntityServer::sender()
        .send(StoreMessage::Command(cmd))
        .await
        .map_err(|_| StoreError::Unavailable)
}
```

pub async fn name(&self) -> Result<&str, LoadError> {

}

// For Role — load all fields (single-asset entity)
async fn ensure_mutable(&mut self) -> Result<(), SetterError> {
    let response = request(StoreRequest::Load {
        any_ref: AnyEntityRef::Role(self.entity_ref.clone()),
        fields: vec![],  // &[] means all fields
    }).await.map_err(|_| SetterError::Substrate(SubstrateError {
        path: self.entity_ref.id().to_string(),
        message: "failed to load entity".to_string(),

    if let StoreResponse::Entity(StoreEntity::Role(loaded)) = response {
        loaded.initialize_into(self);
    }
    Ok(())
}

fn make_role(id: &str) -> Role {
    Role {
        entity_ref:  EntityRef::new(id),
        name:        format!("{} Name", id),
        description: None,
        purpose:     "Purpose".to_string(),
        traits:      None,
        extensions:  HashMap::new(),
    }
}

fn role_any_ref(id: &str) -> AnyEntityRef {
    AnyEntityRef::Role(EntityRef::new(id))
}

// --- insert and resolve ---

#[tokio::test]
async fn insert_then_resolve_returns_entity() {
    EntityServer::with_test(InMemorySubstrate::new(), || async {
        let role = make_role("eng-lead");
        let ref_ = EntityRef::new("eng-lead");
        EntityClient::insert(pari::entity::StoreEntity::from_role(TrackedRole::from(role))).await.unwrap();

        let resolved = EntityClient::resolve(role_any_ref("eng-lead")).await.unwrap();
        assert_eq!(resolved.any_ref().id(), "eng-lead");
    }).await;
}

#[tokio::test]
async fn checkout_then_commit_updates_entity() {
    EntityServer::with_test(InMemorySubstrate::new(), || async {
        // Insert a role
        let role = make_role("eng-lead");
        EntityClient::insert(pari::entity::StoreEntity::from_role(TrackedRole::from(role))).await.unwrap();

        // Checkout
        let mut entity = EntityClient::checkout(role_any_ref("eng-lead")).await.unwrap();

        // Mutate — simulate setter COW replacement
        if let pari::entity::StoreEntity::Role(ref mut r) = entity {
            r.name = std::sync::Arc::new(pari::tracked::TrackedField::with_value("New Name".to_string()));
        }

        // Commit — should merge dirty field into store
        entity.commit().await.unwrap();

        // Resolve — verify committed value
        let resolved = EntityClient::resolve(role_any_ref("eng-lead")).await.unwrap();
        if let pari::entity::StoreEntity::Role(r) = resolved {
            assert_eq!(r.name.get(), Some(&"New Name".to_string()));
        }
    }).await;
}
